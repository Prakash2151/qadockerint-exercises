
#start qwiklabs environment
#login
#create key pair called classkey
#create ppk from this using puttygen
#put in same folder as the build-class

#get access keys from qwiklabs console and configure command line tool
aws configure

#generate template
build-class -d -t class.template -s 5

#get cloud fomation started
aws cloudformation create-stack --stack-name dockerclass --template-body "file://class.template" --parameters "ParameterKey=KeyName,ParameterValue=classkey"

#get the description of the cloud
aws cloudformation describe-stacks --stack-name dockerclass > stackoutput.txt

#parse the outputs to generate individual files
parse-outputs -i stackoutput.txt
